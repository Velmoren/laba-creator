# ОТЧЕТ ПО ЛАБОРАТОРНОЙ РАБОТЕ №5: ПРОГРАММИРОВАНИЕ СЕРВЕРНОЙ ЛОГИКИ С ИСПОЛЬЗОВАНИЕМ КУРСОРОВ

## ВВЕДЕНИЕ
Процедурный язык PL/pgSQL расширяет возможности SQL, позволяя реализовывать сложную бизнес-логику на стороне сервера. Целью работы является создание процедуры автоматизированной вставки клиента с использованием механизмов итерации по наборам данных (курсоров).

## 1. РАЗРАБОТКА ПРОЦЕДУРЫ CUSTOMERINSERT
Процедура выполняет две задачи: регистрирует нового клиента и автоматически привязывает его ко всем художникам определенной национальности, к которым он проявил интерес.

```sql
CREATE OR REPLACE PROCEDURE CustomerInsert (
    newName IN varchar,
    newAreaCode IN varchar,
    newPhone IN varchar,
    artistNationality IN varchar
) AS $$
DECLARE
    artistRecord RECORD;
    artistCursor CURSOR FOR 
        SELECT ArtistID FROM ARTIST WHERE Nationality = artistNationality;
    v_count int;
BEGIN
    -- Проверка на дубликаты
    SELECT count(*) INTO v_count FROM CUSTOMER 
    WHERE Name = newName AND PhoneNumber = newPhone;
    
    IF v_count > 0 THEN
        RAISE NOTICE 'Клиент % уже существует', newName;
        RETURN;
    END IF;

    -- Вставка клиента
    INSERT INTO CUSTOMER(Name, AreaCode, PhoneNumber)
    VALUES (newName, newAreaCode, newPhone);

    -- Работа с курсором
    FOR artistRecord IN artistCursor LOOP
        INSERT INTO CUSTOMER_ARTIST_INT(ArtistID, CustomerID)
        VALUES (artistRecord.ArtistID, currval('CustID_Seq'));
    END LOOP;
END;
$$ LANGUAGE plpgsql;
```

## ЗАКЛЮЧЕНИЕ
Использование курсоров позволило реализовать массовую вставку связей за один вызов процедуры. Это снижает сетевой трафик между клиентом и сервером и инкапсулирует логику внутри БД.
