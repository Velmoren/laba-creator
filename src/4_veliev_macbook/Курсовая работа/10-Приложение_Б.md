**PL/pgSQL Триггеры и Представления**

```sql
-- Триггер: Предотвращение пространственных коллизий в стойке
CREATE OR REPLACE FUNCTION check_rack_collision() RETURNS TRIGGER AS $$
DECLARE
    collision_count INTEGER;
BEGIN
    -- Подсчитываем, есть ли пересекающиеся юниты в той же стойке
    SELECT COUNT(*) INTO collision_count
    FROM PLACEMENT
    WHERE rack_id = NEW.rack_id
      AND (NEW.u_start <= u_end AND NEW.u_end >= u_start)
      AND id != COALESCE(NEW.id, 0); 

    IF collision_count > 0 THEN
        RAISE EXCEPTION 'Конфликт размещения: Юниты с % по % в стойке % уже заняты', NEW.u_start, NEW.u_end, NEW.rack_id;
    END IF;

    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER trg_check_rack_collision
BEFORE INSERT OR UPDATE ON PLACEMENT
FOR EACH ROW EXECUTE FUNCTION check_rack_collision();

-- Представление: Статистика емкости дата-центра
CREATE VIEW Статистика_по_стойкам AS
SELECT 
    r.rack_label AS "Номер_стойки",
    dh.hall_number AS "Машинный_зал",
    r.capacity_u AS "Всего_юнитов",
    COALESCE(SUM(e.size_u), 0) AS "Занято_юнитов",
    r.capacity_u - COALESCE(SUM(e.size_u), 0) AS "Свободно_юнитов"
FROM RACK r
JOIN DATA_HALL dh ON r.hall_id = dh.id
LEFT JOIN PLACEMENT p ON r.id = p.rack_id
LEFT JOIN EQUIPMENT e ON p.equipment_id = e.id
GROUP BY r.id, r.rack_label, dh.hall_number, r.capacity_u;
```
