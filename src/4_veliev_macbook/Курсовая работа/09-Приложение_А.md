# ПРИЛОЖЕНИЕ А

**DDL Скрипты (Схема базы данных)**

```sql
-- Создание справочника тарифных планов
CREATE TABLE TARIFF (
    id SERIAL PRIMARY KEY,
    name VARCHAR(100) NOT NULL UNIQUE,
    power_limit_watts INTEGER NOT NULL,
    bandwidth_mbps INTEGER NOT NULL,
    monthly_cost DECIMAL(10, 2) NOT NULL
);

-- Регистрация клиентов дата-центра
CREATE TABLE CLIENT (
    id SERIAL PRIMARY KEY,
    company_name VARCHAR(150) NOT NULL,
    inn VARCHAR(12) UNIQUE NOT NULL,
    contact_phone VARCHAR(20),
    email VARCHAR(100)
);

-- Договоры на обслуживание
CREATE TABLE CONTRACT (
    id SERIAL PRIMARY KEY,
    client_id INTEGER REFERENCES CLIENT(id) ON DELETE RESTRICT,
    tariff_id INTEGER REFERENCES TARIFF(id) ON DELETE RESTRICT,
    start_date DATE NOT NULL,
    end_date DATE
);

-- Инфраструктура: Машинные залы
CREATE TABLE DATA_HALL (
    id SERIAL PRIMARY KEY,
    hall_number VARCHAR(10) UNIQUE NOT NULL,
    cooling_system VARCHAR(100),
    max_power_kw INTEGER NOT NULL
);

-- Инфраструктура: Стойки
CREATE TABLE RACK (
    id SERIAL PRIMARY KEY,
    hall_id INTEGER REFERENCES DATA_HALL(id) ON DELETE RESTRICT,
    rack_label VARCHAR(50) UNIQUE NOT NULL,
    capacity_u INTEGER NOT NULL CHECK (capacity_u > 0),
    allocated_power_watts INTEGER NOT NULL
);

-- Оборудование клиентов и провайдера
CREATE TABLE EQUIPMENT (
    id SERIAL PRIMARY KEY,
    contract_id INTEGER REFERENCES CONTRACT(id) ON DELETE SET NULL,
    serial_number VARCHAR(100) UNIQUE NOT NULL,
    eq_type VARCHAR(50) NOT NULL,
    manufacturer VARCHAR(100),
    model VARCHAR(100),
    size_u INTEGER NOT NULL CHECK (size_u > 0),
    status VARCHAR(20) NOT NULL DEFAULT 'Available'
);

-- Физическое размещение оборудования в стойках
CREATE TABLE PLACEMENT (
    id SERIAL PRIMARY KEY,
    equipment_id INTEGER REFERENCES EQUIPMENT(id) ON DELETE CASCADE,
    rack_id INTEGER REFERENCES RACK(id) ON DELETE CASCADE,
    u_start INTEGER NOT NULL CHECK (u_start > 0),
    u_end INTEGER NOT NULL CHECK (u_end >= u_start),
    install_date DATE NOT NULL DEFAULT CURRENT_DATE
);

-- Персонал инженерной службы
CREATE TABLE EMPLOYEE (
    id SERIAL PRIMARY KEY,
    full_name VARCHAR(150) NOT NULL,
    position VARCHAR(100),
    phone VARCHAR(20)
);

-- Тикеты (Инциденты)
CREATE TABLE INCIDENT (
    id SERIAL PRIMARY KEY,
    equipment_id INTEGER REFERENCES EQUIPMENT(id) ON DELETE CASCADE,
    created_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    description TEXT NOT NULL,
    severity VARCHAR(20) NOT NULL,
    status VARCHAR(20) NOT NULL DEFAULT 'Open'
);

-- Ассоциативная таблица: Назначение инженеров на тикеты
CREATE TABLE INCIDENT_ASSIGNMENT (
    incident_id INTEGER REFERENCES INCIDENT(id) ON DELETE CASCADE,
    employee_id INTEGER REFERENCES EMPLOYEE(id) ON DELETE CASCADE,
    assigned_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    PRIMARY KEY (incident_id, employee_id)
);
```
