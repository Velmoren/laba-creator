# Приложение А. Скрипт создания базы данных

```sql
-- Создание базы данных
CREATE DATABASE "КомпьютерныеКурсы"
    WITH 
    OWNER = postgres
    ENCODING = 'UTF8'
    LC_COLLATE = 'Russian_Russia.1251'
    LC_CTYPE = 'Russian_Russia.1251'
    TEMPLATE = template0;

-- Подключение к базе данных
\c "КомпьютерныеКурсы"

-- Таблица Клиент
CREATE TABLE Клиент (
    id_клиента SERIAL PRIMARY KEY,
    фамилия VARCHAR(50) NOT NULL,
    имя VARCHAR(50) NOT NULL,
    отчество VARCHAR(50),
    дата_рождения DATE,
    телефон VARCHAR(20),
    email VARCHAR(100),
    адрес TEXT,
    дата_регистрации DATE DEFAULT CURRENT_DATE
);

-- Таблица Преподаватель
CREATE TABLE Преподаватель (
    id_преподавателя SERIAL PRIMARY KEY,
    фамилия VARCHAR(50) NOT NULL,
    имя VARCHAR(50) NOT NULL,
    отчество VARCHAR(50),
    телефон VARCHAR(20),
    email VARCHAR(100),
    квалификация VARCHAR(100),
    стаж_работы INTEGER
);

-- Таблица Курс
CREATE TABLE Курс (
    id_курса SERIAL PRIMARY KEY,
    название VARCHAR(200) NOT NULL,
    описание TEXT,
    продолжительность_часов INTEGER NOT NULL,
    стоимость DECIMAL(10,2) NOT NULL CHECK (стоимость > 0),
    уровень_сложности VARCHAR(30)
);

-- Таблица Группа
CREATE TABLE Группа (
    id_группы SERIAL PRIMARY KEY,
    название VARCHAR(100) NOT NULL,
    id_курса INTEGER NOT NULL REFERENCES Курс(id_курса),
    id_преподавателя INTEGER NOT NULL REFERENCES Преподаватель(id_преподавателя),
    дата_начала DATE,
    дата_конца DATE,
    максимальный_размер INTEGER DEFAULT 15,
    статус VARCHAR(30) DEFAULT 'Формируется'
);

-- Таблица Занятие
CREATE TABLE Занятие (
    id_занятия SERIAL PRIMARY KEY,
    id_группы INTEGER NOT NULL REFERENCES Группа(id_группы),
    номер_занятия INTEGER NOT NULL,
    дата_занятия DATE NOT NULL,
    время_начала TIME NOT NULL,
    время_конца TIME NOT NULL,
    аудитория VARCHAR(20),
    тема_занятия VARCHAR(200)
);

-- Таблица Договор
CREATE TABLE Договор (
    id_договора SERIAL PRIMARY KEY,
    номер_договора VARCHAR(50) NOT NULL UNIQUE,
    id_клиента INTEGER NOT NULL REFERENCES Клиент(id_клиента),
    id_группы INTEGER NOT NULL REFERENCES Группа(id_группы),
    дата_заключения DATE NOT NULL,
    сумма DECIMAL(10,2) NOT NULL,
    статус VARCHAR(30) DEFAULT 'Действует'
);

-- Таблица Посещаемость
CREATE TABLE Посещаемость (
    id_посещаемости SERIAL PRIMARY KEY,
    id_клиента INTEGER NOT NULL REFERENCES Клиент(id_клиента),
    id_занятия INTEGER NOT NULL REFERENCES Занятие(id_занятия),
    статус VARCHAR(20) DEFAULT 'Присутствовал',
    причина VARCHAR(100),
    UNIQUE(id_клиента, id_занятия)
);

-- Таблица Успеваемость
CREATE TABLE Успеваемость (
    id_успеваемости SERIAL PRIMARY KEY,
    id_клиента INTEGER NOT NULL REFERENCES Клиент(id_клиента),
    id_занятия INTEGER NOT NULL REFERENCES Занятие(id_занятия),
    оценка INTEGER CHECK (оценка BETWEEN 1 AND 5),
    домашнее_задание TEXT,
    проверено BOOLEAN DEFAULT FALSE,
    UNIQUE(id_клиента, id_занятия)
);

-- Таблица Оплата
CREATE TABLE Оплата (
    id_оплаты SERIAL PRIMARY KEY,
    id_договора INTEGER NOT NULL REFERENCES Договор(id_договора),
    дата_оплаты DATE NOT NULL,
    сумма DECIMAL(10,2) NOT NULL,
    способ_оплаты VARCHAR(30),
    статус VARCHAR(30) DEFAULT 'Оплачен'
);

-- Таблица Зачисление
CREATE TABLE Зачисление (
    id_зачисления SERIAL PRIMARY KEY,
    id_клиента INTEGER NOT NULL REFERENCES Клиент(id_клиента),
    id_группы INTEGER NOT NULL REFERENCES Группа(id_группы),
    дата_зачисления DATE NOT NULL,
    дата_отчисления DATE,
    статус_обучения VARCHAR(30) DEFAULT 'Обучается',
    UNIQUE(id_клиента, id_группы)
);

-- Таблица Сертификат
CREATE TABLE Сертификат (
    id_сертификата SERIAL PRIMARY KEY,
    номер_сертификата VARCHAR(50) NOT NULL UNIQUE,
    id_клиента INTEGER NOT NULL REFERENCES Клиент(id_клиента),
    id_курса INTEGER NOT NULL REFERENCES Курс(id_курса),
    дата_выдачи DATE,
    оценка VARCHAR(20)
);

-- Создание индексов
CREATE INDEX idx_клиент_фамилия ON Клиент(фамилия);
CREATE INDEX idx_клиент_email ON Клиент(email);
CREATE INDEX idx_курс_название ON Курс(название);
CREATE INDEX idx_группа_курс ON Группа(id_курса);
CREATE INDEX idx_группа_преподаватель ON Группа(id_преподавателя);
CREATE INDEX idx_занятие_группа ON Занятие(id_группы);
CREATE INDEX idx_договор_клиент ON Договор(id_клиента);
CREATE INDEX idx_посещаемость_клиент ON Посещаемость(id_клиента);
CREATE INDEX idx_успеваемость_клиент ON Успеваемость(id_клиента);
CREATE INDEX idx_оплата_договор ON Оплата(id_договора);

-- Создание представлений
CREATE VIEW КлиентыСДоговорами AS
SELECT 
    к.фамилия || ' ' || к.имя || ' ' || COALESCE(к.отчество, '') AS ФИО,
    к.телефон,
    к.email,
    д.номер_договора,
    д.дата_заключения,
    д.сумма AS сумма_договора,
    д.статус AS статус_договора
FROM Клиент к
LEFT JOIN Договор д ON к.id_клиента = д.id_клиента
ORDER BY к.фамилия, к.имя;

CREATE VIEW РасписаниеГруппы AS
SELECT 
    г.название AS группа,
    к.название AS курс,
    п.фамилия || ' ' || п.имя AS преподаватель,
    з.номер_занятия,
    з.дата_занятия,
    з.время_начала,
    з.время_конца,
    з.аудитория,
    з.тема_занятия
FROM Занятие з
JOIN Группа г ON з.id_группы = г.id_группы
JOIN Курс к ON г.id_курса = к.id_курса
JOIN Преподаватель п ON г.id_преподавателя = п.id_преподавателя
ORDER BY з.дата_занятия, з.время_начала;

CREATE VIEW УспеваемостьГруппы AS
SELECT 
    г.название AS группа,
    к.название AS курс,
    клиент.фамилия || ' ' || клиент.имя AS слушатель,
    з.номер_занятия,
    з.тема_занятия,
    у.оценка,
    у.домашнее_задание,
    у.проверено
FROM Успеваемость у
JOIN Занятие з ON у.id_занятия = з.id_занятия
JOIN Группа г ON з.id_группы = г.id_группы
JOIN Курс к ON г.id_курса = к.id_курса
JOIN Клиент клиент ON у.id_клиента = клиент.id_клиента
ORDER BY г.название, з.номер_занятия;

CREATE VIEW Задолженности AS
SELECT 
    к.id_клиента,
    к.фамилия || ' ' || к.имя AS должник,
    д.номер_договора,
    д.сумма AS сумма_договора,
    COALESCE(SUM(оп.сумма), 0) AS оплачено,
    д.сумма - COALESCE(SUM(оп.сумма), 0) AS задолженность
FROM Клиент к
JOIN Договор д ON к.id_клиента = д.id_клиента
LEFT JOIN Оплата оп ON д.id_договора = оп.id_договора
WHERE д.статус = 'Действует'
GROUP BY к.id_клиента, к.фамилия, к.имя, д.id_договора, д.номер_договора, д.сумма
HAVING д.сумма - COALESCE(SUM(оп.сумма), 0) > 0
ORDER BY задолженность DESC;

CREATE VIEW СтатистикаКурсов AS
SELECT 
    к.id_курса,
    к.название,
    к.продолжительность_часов,
    к.стоимость,
    COUNT(DISTINCT г.id_группы) AS количество_групп,
    COUNT(DISTINCT д.id_клиента) AS количество_слушателей,
    COUNT(DISTINCT ce.id_сертификата) AS выдано_сертификатов
FROM Курс к
LEFT JOIN Группа г ON к.id_курса = г.id_курса
LEFT JOIN Договор д ON г.id_группы = д.id_группы
LEFT JOIN Сертификат ce ON к.id_курса = ce.id_курса
GROUP BY к.id_курса, к.название, к.продолжительность_часов, к.стоимость
ORDER BY количество_слушателей DESC;
```
