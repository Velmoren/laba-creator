## Триггер 1: Автоматический расчёт общей суммы оплат по договору

```sql
-- Функция для автоматического обновления статуса оплаты
CREATE OR REPLACE FUNCTION fn_обновить_статус_оплаты()
RETURNS TRIGGER AS $$
BEGIN
    IF NEW.сумма > 0 THEN
        UPDATE Оплата
        SET статус = 'Оплачен'
        WHERE id_оплаты = NEW.id_оплаты;
    END IF;
    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

-- Создание триггера
CREATE TRIGGER trg_обновить_статус_оплаты
AFTER INSERT ON Оплата
FOR EACH ROW EXECUTE FUNCTION fn_обновить_статус_оплаты();
```

**Назначение:** Автоматически устанавливает статус «Оплачен» при добавлении новой оплаты с положительной суммой.

**Время срабатывания:** AFTER INSERT

## Триггер 2: Проверка доступности мест в группе

```sql
-- Функция для проверки наличия свободных мест
CREATE OR REPLACE FUNCTION fn_проверить_место_в_группе()
RETURNS TRIGGER AS $$
DECLARE
    текущее_количество INTEGER;
    максимум_группы INTEGER;
BEGIN
    SELECT COUNT(*), г.максимальный_размер
    INTO текущее_количество, максимум_группы
    FROM Договор д
    JOIN Группа г ON д.id_группы = г.id_группы
    WHERE д.id_группы = NEW.id_группы AND д.статус = 'Действует'
    GROUP BY г.максимальный_размер;
    
    IF текущее_количество >= максимум_группы THEN
        RAISE EXCEPTION 'В группе нет свободных мест! Максимум: %', максимум_группы;
    END IF;
    
    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

-- Создание триггера
CREATE TRIGGER trg_проверить_место_в_группе
BEFORE INSERT ON Договор
FOR EACH ROW EXECUTE FUNCTION fn_проверить_место_в_группе();
```

**Назначение:** Проверяет наличие свободных мест в группе перед добавлением нового договора. Если группа заполнена, выбрасывается исключение.

**Время срабатывания:** BEFORE INSERT

## Триггер 3: Автоматическое закрытие группы при окончании дат

```sql
-- Функция для автоматического закрытия завершённых групп
CREATE OR REPLACE FUNCTION fn_обновить_статус_группы()
RETURNS TRIGGER AS $$
BEGIN
    UPDATE Группа
    SET статус = 'Завершена'
    WHERE дата_конца < CURRENT_DATE AND статус = 'В процессе';
    
    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

-- Создание триггера
CREATE TRIGGER trg_обновить_статус_группы
AFTER INSERT ON Занятие
FOR EACH ROW EXECUTE FUNCTION fn_обновить_статус_группы();
```

**Назначение:** Автоматически изменяет статус группы на «Завершена» при наступлении даты окончания обучения.

**Время срабатывания:** AFTER INSERT

## Триггер 4: Логирование изменений в таблице клиентов

```sql
-- Таблица для хранения истории изменений
CREATE TABLE ИсторияИзмененийКлиентов (
    id_изменения SERIAL PRIMARY KEY,
    id_клиента INTEGER,
    действие VARCHAR(10),
    фамилия_старая VARCHAR(50),
    фамилия_новая VARCHAR(50),
    имя_старое VARCHAR(50),
    имя_новое VARCHAR(50),
    дата_изменения TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- Функция для логирования изменений
CREATE OR REPLACE FUNCTION fn_логировать_изменения_клиента()
RETURNS TRIGGER AS $$
BEGIN
    IF OLD IS NOT NULL THEN
        INSERT INTO ИсторияИзмененийКлиентов 
            (id_клиента, действие, фамилия_старая, фамилия_новая, имя_старое, имя_новое)
        VALUES (
            NEW.id_клиента,
            'UPDATE',
            OLD.фамилия,
            NEW.фамилия,
            OLD.имя,
            NEW.имя
        );
    END IF;
    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

-- Создание триггера
CREATE TRIGGER trg_логировать_изменения_клиента
AFTER UPDATE ON Клиент
FOR EACH ROW EXECUTE FUNCTION fn_логировать_изменения_клиента();
```

**Назначение:** Записывает в журнал все изменения данных о клиентах (ФИО) для обеспечения аудита.

**Время срабатывания:** AFTER UPDATE

## Ограничения CHECK

```sql
-- Проверка положительной стоимости курса
ALTER TABLE Курс
ADD CONSTRAINT CK_стоимость_курса CHECK (стоимость > 0);

-- Проверка корректности оценки
ALTER TABLE Успеваемость
ADD CONSTRAINT CK_оценка CHECK (оценка BETWEEN 1 AND 5);

-- Проверка положительной суммы оплаты
ALTER TABLE Оплата
ADD CONSTRAINT CK_сумма_оплаты CHECK (сумма > 0);

-- Проверка статуса договора
ALTER TABLE Договор
ADD CONSTRAINT CK_статус_договора 
CHECK (статус IN ('Действует', 'Расторгнут', 'Завершён'));

-- Проверка статуса посещаемости
ALTER TABLE Посещаемость
ADD CONSTRAINT CK_посещаемость_статус
CHECK (статус IN ('Присутствовал', 'Отсутствовал', 'Болен'));
```
